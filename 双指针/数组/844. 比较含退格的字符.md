    class Solution {
    public:
        bool backspaceCompare(string s, string t) {
            int skipS = 0, skipT = 0;
            int i = s.size()-1, j = t.size()-1;
            while(i >= 0 || j >= 0){
                while(i >= 0){
                    if(s[i] == '#'){
                        skipS++;
                        i--;
                    }
                    else if(skipS != 0){
                        i--;
                        skipS--;
                    }
                    else{
                        break;
                    }
                }
                while( j >= 0){
                    if(t[j] == '#'){
                        skipT++;
                        j--;
                    }
                    else if(skipT != 0){
                        j--;
                        skipT--;
                    }
                    else{
                        break;
                    }
                }
                if(i >= 0 && j >= 0){
                    if(s[i] != t[j]) return false;
                }
                else{
                    if(i >=0 || j >= 0){
                        return false;
                    }
                }
                i--;
                j--;
            }
            return true;
        }
    };
# Solution
  * 双指针的方法进行比对两个字符串，因为 # 删除的是前面的空格，所以这里是 从后往前 进行遍历
  * 设置 skip 统计 # 的个数，移动相关数量的指针
